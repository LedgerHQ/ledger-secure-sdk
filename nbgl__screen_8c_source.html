306 int nbgl_screenPush(nbgl_obj_t ***elements,
307  uint8_t nbElements,
308  const nbgl_screenTickerConfiguration_t *ticker,
309 #ifdef HAVE_SE_TOUCH
310  nbgl_touchCallback_t callback)
311 {
312 #else // HAVE_SE_TOUCH
313  nbgl_buttonCallback_t callback)
314 {
315 #endif // HAVE_SE_TOUCH
316  uint8_t screenIndex;
317  if (nbScreensOnStack >= SCREEN_STACK_SIZE) {
318  LOG_WARN(SCREEN_LOGGER,
319  "nbgl_screenPush(): already in highest index in the stack(%d)\n",
320  nbScreensOnStack - 1);
321  return -1;
322  }
323  // if no screen, consider it as a first fake push
324  if (nbScreensOnStack == 0) {
325  screenIndex = 1; // push at position 1 because 0 is reserved for background
326  topOfStack = &screenStack[screenIndex];
327  topOfStack->next = NULL;
328  // link top of stack to background (even if empty)
329  topOfStack->previous = &screenStack[0];
330  screenStack[0].next = topOfStack;
331  screenStack[0].container.nbChildren = 0;
332  // count empty background as an active screen
333  nbScreensOnStack++;
334  }
335  else {
336  // find a non used screen in the array
337  for (screenIndex = 1; screenIndex < SCREEN_STACK_SIZE; screenIndex++) {
338  if (screenStack[screenIndex].previous == NULL) {
339  // if no previous, means unused, so take it
340  // update previous topOfStack
341  topOfStack->next = &screenStack[screenIndex];
342  screenStack[screenIndex].previous = topOfStack;
343  // new top of stack
344  topOfStack = &screenStack[screenIndex];
345  topOfStack->next = NULL;
346  break;
347  }
348  }
349  if (screenIndex == SCREEN_STACK_SIZE) {
350  // should never happen
351  LOG_WARN(SCREEN_LOGGER, "nbgl_screenPush(): corruption in stack\n");
352  }
353  }
354  if (nbgl_screenSetAt(screenIndex, elements, nbElements, ticker, callback) >= 0) {
355  nbScreensOnStack++;
356  LOG_DEBUG(SCREEN_LOGGER, "nbgl_screenPush(): screen %d is now top of stack\n", screenIndex);
357  return screenIndex;
358  }
359  else {
360  return -1;
361  }
362 }

373 int nbgl_screenPop(uint8_t screenIndex)
374 {
375  if (nbScreensOnStack == 0) {
376  LOG_WARN(SCREEN_LOGGER, "nbgl_screenPop(): already in lowest index in the stack\n");
377  return -1;
378  }
379  LOG_DEBUG(SCREEN_LOGGER, "nbgl_screenPop(): at index %d\n", screenIndex);
380  nbScreensOnStack--;
381  // move top of stack if needed
382  if (&screenStack[screenIndex] == topOfStack) {
383  if (nbScreensOnStack == 0) {
384  topOfStack = NULL;
385  }
386  else {
387  topOfStack = topOfStack->previous;
388  }
389  }
390  else {
391  // connect previous to next
392  if (screenStack[screenIndex].previous != NULL) {
393  screenStack[screenIndex].previous->next = screenStack[screenIndex].next;
394  }
395  if (screenStack[screenIndex].next != NULL) {
396  screenStack[screenIndex].next->previous = screenStack[screenIndex].previous;
397  }
398  }
399  // free slot
400  screenStack[screenIndex].previous = NULL;
401  screenStack[screenIndex].next = NULL;
402  screenStack[screenIndex].container.nbChildren = 0;
403  screenStack[screenIndex].container.children = NULL;
404  // release used objects and containers
405  nbgl_objPoolRelease(screenIndex);
406  nbgl_containerPoolRelease(screenIndex);
407 
408  // special case when we pop the only modal and no real background is under it
409  if ((nbScreensOnStack == 1) && (screenStack[0].container.nbChildren == 0)) {
410  nbScreensOnStack = 0;
411  topOfStack = NULL;
412  }
413  return 0;
414 }

446 {
