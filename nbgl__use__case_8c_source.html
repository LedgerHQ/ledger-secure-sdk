691 // Helper that does the computing of which nbgl_content_t and which element in the content should be
692 // displayed for the next generic context navigation page
693 static const nbgl_content_t *genericContextComputeNextPageParams(uint8_t pageIdx,
694  nbgl_content_t *content,
695  uint8_t *p_nbElementsInNextPage,
696  bool *p_flag)
697 {
698  int8_t nextContentIdx = genericContext.currentContentIdx;
699  int16_t nextElementIdx = genericContext.currentElementIdx;
700  uint8_t nbElementsInNextPage;
701 
702  // Retrieve info on the next page
703  genericContextGetPageInfo(pageIdx, &nbElementsInNextPage, p_flag);
704  *p_nbElementsInNextPage = nbElementsInNextPage;
705 
706  // Handle forward navigation:
707  // add to current index the number of pairs of the currently active page
708  if (pageIdx > navInfo.activePage) {
709  uint8_t nbElementsInCurrentPage;
710 
711  genericContextGetPageInfo(navInfo.activePage, &nbElementsInCurrentPage, NULL);
712  nextElementIdx += nbElementsInCurrentPage;
713 
714  // Handle case where the content to be displayed is in the next content
715  // In such case start at element index 0.
716  if (nextElementIdx >= genericContext.currentContentElementNb) {
717  nextContentIdx += 1;
718  nextElementIdx = 0;
719  }
720  }
721 
722  // Handle backward navigation:
723  // add to current index the number of pairs of the currently active page
724  if (pageIdx < navInfo.activePage) {
725  // Backward navigation: remove to current index the number of pairs of the current page
726  nextElementIdx -= nbElementsInNextPage;
727 
728  // Handle case where the content to be displayed is in the previous content
729  // In such case set a negative number as element index so that it is handled
730  // later once the previous content is accessible so that its elements number
731  // can be retrieved.
732  if (nextElementIdx < 0) {
733  nextContentIdx -= 1;
734  nextElementIdx = -nbElementsInNextPage;
735  }
736  }
737 
738  const nbgl_content_t *p_content;
739  // Retrieve next content
740  if ((nextContentIdx == -1) && (genericContext.hasStartingContent)) {
741  p_content = &STARTING_CONTENT;
742  }
743  else if ((nextContentIdx == genericContext.genericContents.nbContents)
744  && (genericContext.hasFinishingContent)) {
745  p_content = &FINISHING_CONTENT;
746  }
747  else {
748  p_content = getContentAtIdx(&genericContext.genericContents, nextContentIdx, content);
749 
750  if (p_content == NULL) {
751  LOG_DEBUG(USE_CASE_LOGGER, "Fail to retrieve content\n");
752  return NULL;
753  }
754  }
755 
756  // Handle cases where we are going to display data from a new content:
757  // - navigation to a previous or to the next content
758  // - First page display (genericContext.currentContentElementNb == 0)
759  //
760  // In such case we need to:
761  // - Update genericContext.currentContentIdx
762  // - Update genericContext.currentContentElementNb
763  // - Update onContentAction callback
764  // - Update nextElementIdx in case it was previously not calculable
765  if ((nextContentIdx != genericContext.currentContentIdx)
766  || (genericContext.currentContentElementNb == 0)) {
767  genericContext.currentContentIdx = nextContentIdx;
768  genericContext.currentContentElementNb = getContentNbElement(p_content);
769  onContentAction = PIC(p_content->contentActionCallback);
770  if (nextElementIdx < 0) {
771  nextElementIdx = genericContext.currentContentElementNb + nextElementIdx;
772  }
773  }
774 
775  // Sanity check
776  if ((nextElementIdx < 0) || (nextElementIdx >= genericContext.currentContentElementNb)) {
777  LOG_DEBUG(USE_CASE_LOGGER,
778  "Invalid element index %d / %d\n",
779  nextElementIdx,
780  genericContext.currentContentElementNb);
781  return NULL;
782  }
783 
784  // Update genericContext elements
785  genericContext.currentElementIdx = nextElementIdx;
786  navInfo.activePage = pageIdx;
787 
788  return p_content;
789 }
