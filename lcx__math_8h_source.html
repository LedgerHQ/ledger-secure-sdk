569static inline bool cx_math_is_zero(const uint8_t *a, size_t len)
570{
571 uint8_t acc = 0; // accumulate all the bytes in order to run in constant time
572 for (size_t i = 0; i < len; i++) {
573 acc |= a[i];
574 }
575 return acc == 0;
576}
