26 bool swap_str_to_u64(const uint8_t *src, size_t length, uint64_t *result)
27 {
28  if (length > sizeof(uint64_t)) {
29  return false;
30  }
31  uint64_t value = 0;
32  for (size_t i = 0; i < length; i++) {
33  value <<= 8;
34  value |= src[i];
35  }
36  *result = value;
37  return true;
38 }
