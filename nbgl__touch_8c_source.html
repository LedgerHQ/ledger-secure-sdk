54static void applytouchStatePosition(nbgl_obj_t *obj, nbgl_touchType_t eventType)
55{
56 nbgl_screen_t *screen = (nbgl_screen_t *) nbgl_screenGetTop();
57 if (!obj) {
58 return;
59 }
60 LOG_DEBUG(TOUCH_LOGGER, "Apply event %d on object of type %d\n", eventType, obj->type);
61 /* the first action is the one provided by the application */
62 if ((obj->touchMask & (1 << eventType)) != 0) {
63 // for some specific objects, call directly a specific callback
64 switch (obj->type) {
65#ifdef NBGL_KEYBOARD
66 case KEYBOARD:
67 nbgl_keyboardTouchCallback(obj, eventType);
68 break;
69#endif // NBGL_KEYBOARD
70#ifdef NBGL_KEYPAD
71 case KEYPAD:
72 nbgl_keypadTouchCallback(obj, eventType);
73 break;
74#endif // NBGL_KEYPAD
75 default:
76 if (screen->touchCallback != NULL) {
77 ((nbgl_touchCallback_t) PIC(screen->touchCallback))((void *) obj, eventType);
78 }
79 break;
80 }
81 }
82}

93static nbgl_obj_t *getTouchedObject(nbgl_obj_t *obj, nbgl_touchStatePosition_t *event)
94{
95 if (obj == NULL) {
96 return NULL;
97 }
98 /* check coordinates
99 no need to go further if the touched point is not within the object
100 And because the children are also within the object, no need to check them either */
101 if ((event->x < obj->area.x0) || (event->x >= (obj->area.x0 + obj->area.width))
102 || (event->y < obj->area.y0) || (event->y >= (obj->area.y0 + obj->area.height))) {
103 return NULL;
104 }
105 if ((obj->type == SCREEN) || (obj->type == CONTAINER)) {
106 nbgl_container_t *container = (nbgl_container_t *) obj;
107 // parse the children, if any
108 if (container->children != NULL) {
109 int8_t i = container->nbChildren - 1;
110 // parse children from the latest, because they are drawn from the first
111 while (i >= 0) {
112 nbgl_obj_t *current = container->children[i];
113 if (current != NULL) {
114 current = getTouchedObject(current, event);
115 if (current != NULL) {
116 return current;
117 }
118 }
119 i--;
120 }
121 }
122 }
123 /* now see if the object is interested by touch events (any of them) */
124 if (obj->touchMask != 0) {
125 // LOG_DEBUG(TOUCH_LOGGER,"%d %d \n",clickableObjectTypes ,(1<<obj->type));
126 return obj;
127 }
128 else {
129 return NULL;
130 }
131}

140static nbgl_obj_t *getSwipableObjectAtPos(nbgl_obj_t *obj,
141 nbgl_touchStatePosition_t *pos,
142 nbgl_touchType_t detectedSwipe)
143{
144 if (obj == NULL) {
145 return NULL;
146 }
147
148 // Check if pos position belongs to obj
149 if ((pos->x < obj->area.x0) || (pos->x >= (obj->area.x0 + obj->area.width))
150 || (pos->y < obj->area.y0) || (pos->y >= (obj->area.y0 + obj->area.height))) {
151 return NULL;
152 }
153
154 if ((obj->type == SCREEN) || (obj->type == CONTAINER)) {
155 nbgl_container_t *container = (nbgl_container_t *) obj;
156 for (uint8_t i = 0; i < container->nbChildren; i++) {
157 nbgl_obj_t *current = container->children[i];
158 if (current != NULL) {
159 nbgl_obj_t *child = getSwipableObjectAtPos(current, pos, detectedSwipe);
160 if (child) {
161 return child;
162 }
163 }
164 }
165 }
166 if (obj->touchMask & (1 << detectedSwipe)) {
167 return obj;
168 }
169 return NULL;
170}

186static nbgl_obj_t *getSwipableObject(nbgl_obj_t *obj,
187 nbgl_touchStatePosition_t *first,
188 nbgl_touchStatePosition_t *last,
189 nbgl_touchType_t detectedSwipe)
190{
191 if (obj == NULL) {
192 return NULL;
193 }
194
195 nbgl_obj_t *first_obj = getSwipableObjectAtPos(obj, first, detectedSwipe);
196
197 if (first_obj == NULL) {
198 return NULL;
199 }
200
201 nbgl_obj_t *last_obj = getSwipableObjectAtPos(obj, last, detectedSwipe);
202
203 // Swipable objects match
204 if (first_obj == last_obj) {
205 return first_obj;
206 }
207
208 return NULL;
209}

213#ifndef HAVE_HW_TOUCH_SWIPE
214#define SWIPE_THRESHOLD_X 10
215#define SWIPE_THRESHOLD_Y 20
216#else
217// Mapping between nbgl_hardwareSwipe_t and nbgl_touchEvent_t
218const nbgl_touchType_t SWIPE_GESTURES[] = {[HARDWARE_SWIPE_UP] = SWIPED_UP,
219 [HARDWARE_SWIPE_DOWN] = SWIPED_DOWN,
220 [HARDWARE_SWIPE_LEFT] = SWIPED_LEFT,
221 [HARDWARE_SWIPE_RIGHT] = SWIPED_RIGHT};

224static nbgl_touchType_t nbgl_detectSwipe(nbgl_touchStatePosition_t *last,
225 nbgl_touchStatePosition_t *first)
226{
227#ifdef HAVE_HW_TOUCH_SWIPE
228 // Swipe is detected by hardware
229 (void) first;
230
231 if (last->swipe >= NO_HARDWARE_SWIPE) {
232 return NB_TOUCH_TYPES;
233 }
234
235 return SWIPE_GESTURES[last->swipe];
236
237#else
238 // Swipe is detected by software
239 nbgl_touchType_t detected_swipe = NB_TOUCH_TYPES;
240 if ((last->x - first->x) >= SWIPE_THRESHOLD_X) {
241 detected_swipe = SWIPED_RIGHT;
242 }
243 else if ((first->x - last->x) >= SWIPE_THRESHOLD_X) {
244 detected_swipe = SWIPED_LEFT;
245 }
246 else if ((last->y - first->y) >= SWIPE_THRESHOLD_Y) {
247 detected_swipe = SWIPED_DOWN;
248 }
249 else if ((first->y - last->y) >= SWIPE_THRESHOLD_Y) {
250 detected_swipe = SWIPED_UP;
251 }
252
253 return detected_swipe;
254#endif // HAVE_HW_TOUCH_SWIPE
255}

267void nbgl_touchHandler(nbgl_touchStatePosition_t *touchStatePosition, uint32_t currentTime)
268{
269 static nbgl_touchState_t lastState = RELEASED;
270 nbgl_obj_t *foundObj;
271
272 // save last received currentTime
273 lastCurrentTime = currentTime;
274
275 if (lastState == RELEASED) {
276 // filter out not realistic cases (successive RELEASE events)
277 if (RELEASED == touchStatePosition->state) {
278 lastState = touchStatePosition->state;
279 return;
280 }
281 // memorize first touched position
282 memcpy(&firstTouchedPosition, touchStatePosition, sizeof(nbgl_touchStatePosition_t));
283 }
284 // LOG_DEBUG(TOUCH_LOGGER,"state = %s, x = %d, y=%d\n",(touchStatePosition->state ==
285 // RELEASED)?"RELEASED":"PRESSED",touchStatePosition->x,touchStatePosition->y);
286
287 // parse the whole screen to find proper object
288 foundObj = getTouchedObject(nbgl_screenGetTop(), touchStatePosition);
289
290 // LOG_DEBUG(TOUCH_LOGGER,"nbgl_touchHandler: found obj %p, type = %d\n",foundObj,
291 // foundObj->type);
292 if (foundObj == NULL) {
293 LOG_DEBUG(TOUCH_LOGGER, "nbgl_touchHandler: no found obj\n");
294 if ((touchStatePosition->state == PRESSED) && (lastState == PRESSED)
295 && (lastPressedObj != NULL)) {
296 // finger has moved out of an object
297 // make sure lastPressedObj still belongs to current screen before warning it
298 if (nbgl_screenContainsObj(lastPressedObj)) {
299 applytouchStatePosition(lastPressedObj, OUT_OF_TOUCH);
300 }
301 }
302 // Released event has been handled, forget lastPressedObj
303 lastPressedObj = NULL;
304 }
305
306 // memorize last touched position
307 memcpy(&lastTouchedPosition, touchStatePosition, sizeof(nbgl_touchStatePosition_t));
308
309 if (touchStatePosition->state == RELEASED) {
310 nbgl_touchType_t swipe = nbgl_detectSwipe(touchStatePosition, &firstTouchedPosition);
311 bool consumed = false;
312
313 if (swipe != NB_TOUCH_TYPES) {
314 // Swipe detected
315 nbgl_obj_t *swipedObj = getSwipableObject(
316 nbgl_screenGetTop(), &firstTouchedPosition, &lastTouchedPosition, swipe);
317 // if a swipable object has been found
318 if (swipedObj) {
319 applytouchStatePosition(swipedObj, swipe);
320 consumed = true;
321 }
322 }
323 if (!consumed && (lastPressedObj != NULL)
324 && ((foundObj == lastPressedObj) || (nbgl_screenContainsObj(lastPressedObj)))) {
325 // very strange if lastPressedObj != foundObj, let's consider that it's a normal release
326 // on lastPressedObj make sure lastPressedObj still belongs to current screen before
327 // "releasing" it
328 applytouchStatePosition(lastPressedObj, TOUCH_RELEASED);
329 if (currentTime >= (lastPressedTime + LONG_TOUCH_DURATION)) {
330 applytouchStatePosition(lastPressedObj, LONG_TOUCHED);
331 }
332 else if (currentTime >= (lastPressedTime + SHORT_TOUCH_DURATION)) {
333 applytouchStatePosition(lastPressedObj, TOUCHED);
334 }
335 }
336 // Released event has been handled, forget lastPressedObj
337 lastPressedObj = NULL;
338 }
339 else { // PRESSED
340 if ((lastState == PRESSED) && (lastPressedObj != NULL)) {
341 if (foundObj != lastPressedObj) {
342 // finger has moved out of an object
343 // make sure lastPressedObj still belongs to current screen before warning it
344 if (nbgl_screenContainsObj(lastPressedObj)) {
345 applytouchStatePosition(lastPressedObj, OUT_OF_TOUCH);
346 }
347 lastPressedObj = NULL;
348 }
349 else {
350 // warn the concerned object that it is still touched
351 applytouchStatePosition(foundObj, TOUCHING);
352 }
353 }
354 else if (lastState == RELEASED) {
355 // newly touched object
356 lastPressedObj = foundObj;
357 lastPressedTime = currentTime;
358 applytouchStatePosition(foundObj, TOUCH_PRESSED);
359 applytouchStatePosition(foundObj, TOUCHING);
360 }
361 }
362
363 lastState = touchStatePosition->state;
364}

395nbgl_obj_t *nbgl_touchGetObjectFromId(nbgl_obj_t *obj, uint8_t id)
396{
397 if (obj == NULL) {
398 return NULL;
399 }
400 if ((obj->type == SCREEN) || (obj->type == CONTAINER)) {
401 nbgl_container_t *container = (nbgl_container_t *) obj;
402 // parse the children, if any
403 if (container->children != NULL) {
404 uint8_t i;
405 for (i = 0; i < container->nbChildren; i++) {
406 nbgl_obj_t *current = container->children[i];
407 if (current != NULL) {
408 current = nbgl_touchGetObjectFromId(current, id);
409 if (current != NULL) {
410 return current;
411 }
412 }
413 }
414 }
415 }
416 /* now see if the object is interested by touch events (any of them) */
417 if ((obj->touchMask != 0) && (obj->touchId == id)) {
418 LOG_DEBUG(TOUCH_LOGGER, "found %p: id = %d, type = %d \n", obj, obj->touchId, obj->type);
419 return obj;
420 }
421 else {
422 return NULL;
423 }
424}
