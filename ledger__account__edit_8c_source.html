195static bool handle_hmac_proof(const tlv_data_t *data, s_edit_ledger_account_ctx *context)
196{

199 PRINTF("[Edit Ledger Account] HMAC_PROOF: invalid length (expected %d bytes)\n",

215static bool verify_fields(const s_edit_ledger_account_ctx *context)
216{
217 bool result = TLV_CHECK_RECEIVED_TAGS(context->received_tags,
218 TAG_STRUCTURE_TYPE,
219 TAG_STRUCTURE_VERSION,
220 TAG_CONTACT_NAME,
221 TAG_PREVIOUS_NAME,
222 TAG_DERIVATION_PATH,
223 TAG_BLOCKCHAIN_FAMILY,
224 TAG_HMAC_PROOF);
225 if (!result) {
226 PRINTF("Missing mandatory fields in Edit Ledger Account descriptor!\n");
227 return false;
228 }
229 if (context->edit->ledger_account.blockchain_family == FAMILY_ETHEREUM) {
230 result = TLV_CHECK_RECEIVED_TAGS(context->received_tags, TAG_CHAIN_ID);
231 if (!result) {
232 PRINTF("Missing CHAIN_ID for Ethereum family in Edit Ledger Account descriptor!\n");
233 return false;
234 }
235 }
236 return true;
237}

281static bool build_and_send_response(void)
282{
283 uint8_t hmac_proof[CX_SHA256_SIZE] = {0};
284
285 if (!address_book_compute_hmac_proof_ledger_account(
286 &EDIT_LEDGER_ACCOUNT.ledger_account.bip32_path,
287 (const char *) EDIT_LEDGER_ACCOUNT.ledger_account.account_name,
288 EDIT_LEDGER_ACCOUNT.ledger_account.blockchain_family,
289 EDIT_LEDGER_ACCOUNT.ledger_account.chain_id,
290 hmac_proof)) {
291 PRINTF("[Edit Ledger Account] Error: Failed to compute new HMAC proof\n");
292 return false;
293 }
294
295 bool ok = address_book_send_hmac_proof(TYPE_EDIT_LEDGER_ACCOUNT, hmac_proof);
296 explicit_bzero(hmac_proof, sizeof(hmac_proof));
297 return ok;
298}

305static void review_choice(bool confirm)
306{
307 if (confirm) {
308 if (build_and_send_response()) {
309 nbgl_useCaseStatus("Account name changed", true, finalize_ui_edit_ledger_account);
310 }
311 else {
312 PRINTF("[Edit Ledger Account] Error: Failed to build and send HMAC proof\n");
313 io_send_sw(SWO_INCORRECT_DATA);
314 nbgl_useCaseStatus("Error editing account", false, finalize_ui_edit_ledger_account);
315 }
316 }
317 else {
318 io_send_sw(SWO_INCORRECT_DATA);
319 nbgl_useCaseReviewStatus(STATUS_TYPE_OPERATION_REJECTED, finalize_ui_edit_ledger_account);
320 }
321}

326static void ui_display(void)
327{
328 uint8_t nbPairs = 0;

344 "Edit account name",
345 NULL,

359bolos_err_t edit_ledger_account(uint8_t *buffer_in, size_t buffer_in_length)
360{
361 const buffer_t payload = {.ptr = buffer_in, .size = buffer_in_length};
362 s_edit_ledger_account_ctx ctx = {0};
363
364 // Init the structure
365 ctx.edit = &EDIT_LEDGER_ACCOUNT;
366 memset(&EDIT_LEDGER_ACCOUNT, 0, sizeof(EDIT_LEDGER_ACCOUNT));
367
368 // Parse using SDK TLV parser
369 if (!edit_ledger_account_tlv_parser(&payload, &ctx, &ctx.received_tags)) {
370 PRINTF("[Edit Ledger Account] TLV parsing failed\n");
371 return SWO_INCORRECT_DATA;
372 }
373 if (!verify_fields(&ctx)) {
374 return SWO_INCORRECT_DATA;
375 }
376 print_payload(&ctx);
377
378 // Verify that the host holds a valid proof from the previous registration
379 if (!address_book_verify_hmac_proof_ledger_account(
380 &EDIT_LEDGER_ACCOUNT.ledger_account.bip32_path,
381 EDIT_LEDGER_ACCOUNT.previous_account_name,
382 EDIT_LEDGER_ACCOUNT.ledger_account.blockchain_family,
383 EDIT_LEDGER_ACCOUNT.ledger_account.chain_id,
384 ctx.hmac_proof)) {
385 PRINTF("[Edit Ledger Account] HMAC proof verification failed\n");
386 return SWO_SECURITY_CONDITION_NOT_SATISFIED;
387 }
388
389 // Display confirmation UI
390 ui_display();
391 return SWO_NO_RESPONSE;
392}
