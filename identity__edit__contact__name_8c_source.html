248static bool build_and_send_response(void)
249{
250 uint8_t hmac_name[CX_SHA256_SIZE] = {0};
251
252 if (!address_book_compute_hmac_name(&EDIT_CONTACT_NAME.bip32_path,
253 EDIT_CONTACT_NAME.gid,
254 EDIT_CONTACT_NAME.contact_name,
255 hmac_name)) {
256 PRINTF("[Edit Contact Name] Error: Failed to compute new HMAC_NAME\n");
257 return false;
258 }
259
260 bool ok = address_book_send_hmac_proof(TYPE_EDIT_CONTACT_NAME, hmac_name);
261 explicit_bzero(hmac_name, sizeof(hmac_name));
262 return ok;
263}

270static void review_choice(bool confirm)

324bolos_err_t edit_contact_name(uint8_t *buffer_in, size_t buffer_in_length)
325{
326 const buffer_t payload = {.ptr = buffer_in, .size = buffer_in_length};
327 s_edit_contact_name_ctx ctx = {0};
328
329 // Init the structure
330 ctx.edit = &EDIT_CONTACT_NAME;
331
332 // Parse using SDK TLV parser
333 if (!edit_contact_name_tlv_parser(&payload, &ctx, &ctx.received_tags)) {
334 PRINTF("[Edit Contact Name] TLV parsing failed\n");
335 return SWO_INCORRECT_DATA;
336 }
337 if (!verify_fields(&ctx)) {
338 return SWO_INCORRECT_DATA;
339 }
340 print_payload(&ctx);
341
342 // Verify the group handle and extract the gid
343 if (!address_book_verify_group_handle(
344 &EDIT_CONTACT_NAME.bip32_path, ctx.group_handle, EDIT_CONTACT_NAME.gid)) {
345 PRINTF("[Edit Contact Name] Group handle verification failed\n");
346 return SWO_SECURITY_CONDITION_NOT_SATISFIED;
347 }
348
349 // Verify the wallet holds a valid HMAC_NAME for the previous name
350 if (!address_book_verify_hmac_name(&EDIT_CONTACT_NAME.bip32_path,
351 EDIT_CONTACT_NAME.gid,
352 EDIT_CONTACT_NAME.previous_contact_name,
353 ctx.hmac_proof)) {
354 PRINTF("[Edit Contact Name] HMAC_NAME verification failed\n");
355 return SWO_SECURITY_CONDITION_NOT_SATISFIED;
356 }
357
358 // Display confirmation UI
359 ui_display();
360 return SWO_NO_RESPONSE;
361}
