47bool address_book_handle_derivation_path(const tlv_data_t *data, path_bip32_t *bip32_path)
48{
49 buffer_t path_buffer = {0};
50
51 if (!get_buffer_from_tlv_data(data, &path_buffer, 1, MAX_BIP32_PATH * 4 + 1)) {
52 PRINTF("DERIVATION_PATH: failed to extract\n");
53 return false;
54 }
55
56 // Parse BIP32 path using the standard parser
57 if (buffer_get_path_bip32(&path_buffer, bip32_path) == false) {
58 PRINTF("DERIVATION_PATH: failed to parse BIP32 path\n");
59 return false;
60 }
61
62 return true;
63}

88bool address_book_handle_blockchain_family(const tlv_data_t *data, blockchain_family_e *family)
89{
90 uint8_t value = 0;
91 if (!get_uint8_t_from_tlv_data(data, &value)) {
92 PRINTF("UINT8: failed to extract\n");
93 return false;
94 }
95 if (value >= FAMILY_COUNT) {
96 PRINTF("BLOCKCHAIN_FAMILY: Value out of range (%d, max %d)!\n", value, FAMILY_COUNT - 1);
97 return false;
98 }
99 *family = (blockchain_family_e) value;
100 return true;
101}

113bool address_book_handle_printable_string(const tlv_data_t *data,
114 char *output_buffer,
