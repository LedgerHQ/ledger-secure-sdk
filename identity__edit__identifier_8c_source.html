298static bool verify_fields(const s_edit_ctx *context)
299{
300 bool result = TLV_CHECK_RECEIVED_TAGS(context->received_tags,
301 TAG_STRUCTURE_TYPE,
302 TAG_STRUCTURE_VERSION,
303 TAG_CONTACT_NAME,
304 TAG_ACCOUNT_IDENTIFIER,
305 TAG_PREVIOUS_IDENTIFIER,
306 TAG_GROUP_HANDLE,
307 TAG_DERIVATION_PATH,
308 TAG_BLOCKCHAIN_FAMILY,
309 TAG_HMAC_NAME,
310 TAG_HMAC_REST);
311 if (!result) {
312 PRINTF("[Edit Identifier] Missing mandatory fields!\n");
313 return false;
314 }
315 if (context->edit->identity.blockchain_family == FAMILY_ETHEREUM) {
316 result = TLV_CHECK_RECEIVED_TAGS(context->received_tags, TAG_CHAIN_ID);
317 if (!result) {
318 PRINTF("[Edit Identifier] Missing CHAIN_ID for Ethereum family!\n");
319 return false;
320 }
321 }
322 return true;
323}

357static bool build_and_send_response(void)
358{
359 uint8_t hmac_rest[CX_SHA256_SIZE] = {0};
360
361 if (!address_book_compute_hmac_rest(&EDIT_IDENTIFIER.identity.bip32_path,
362 EDIT_IDENTIFIER.identity.gid,
363 EDIT_IDENTIFIER.identity.scope,
364 EDIT_IDENTIFIER.identity.identifier,
365 EDIT_IDENTIFIER.identity.identifier_len,
366 EDIT_IDENTIFIER.identity.blockchain_family,
367 EDIT_IDENTIFIER.identity.chain_id,
368 hmac_rest)) {
369 PRINTF("[Edit Identifier] Error: Failed to compute new HMAC_REST\n");
370 return false;
371 }
372
373 bool ok = address_book_send_hmac_proof(TYPE_EDIT_IDENTIFIER, hmac_rest);
374 explicit_bzero(hmac_rest, sizeof(hmac_rest));
375 return ok;
376}

383static void review_choice(bool confirm)
384{
385 if (confirm) {
386 if (build_and_send_response()) {
387 if (EDIT_IDENTIFIER.identity.blockchain_family == FAMILY_ETHEREUM) {
388 nbgl_useCaseStatus("Address changed", true, finalize_ui_edit_identifier);
389 }
390 else {
391 nbgl_useCaseStatus("Identifier changed", true, finalize_ui_edit_identifier);
392 }
393 }
394 else {
395 PRINTF("[Edit Identifier] Error: Failed to build and send HMAC proof\n");
396 io_send_sw(SWO_INCORRECT_DATA);
397 nbgl_useCaseStatus("Error during update", false, finalize_ui_edit_identifier);
398 }
399 }
400 else {
401 io_send_sw(SWO_INCORRECT_DATA);
402 nbgl_useCaseReviewStatus(STATUS_TYPE_OPERATION_REJECTED, finalize_ui_edit_identifier);
403 }
404}

434bolos_err_t edit_identifier(uint8_t *buffer_in, size_t buffer_in_length)
435{
436 const buffer_t payload = {.ptr = buffer_in, .size = buffer_in_length};
437 s_edit_ctx ctx = {0};
438
439 // Init the structure
440 ctx.edit = &EDIT_IDENTIFIER;
441
442 // Parse using SDK TLV parser
443 if (!edit_identifier_tlv_parser(&payload, &ctx, &ctx.received_tags)) {
444 PRINTF("[Edit Identifier] TLV parsing failed\n");
445 return SWO_INCORRECT_DATA;
446 }
447 if (!verify_fields(&ctx)) {
448 return SWO_INCORRECT_DATA;
449 }
450 print_payload(&ctx);
451
452 // Verify the group handle and extract the gid
453 if (!address_book_verify_group_handle(
454 &EDIT_IDENTIFIER.identity.bip32_path, ctx.group_handle, EDIT_IDENTIFIER.identity.gid)) {
455 PRINTF("[Edit Identifier] Group handle verification failed\n");
456 return SWO_SECURITY_CONDITION_NOT_SATISFIED;
457 }
458
459 // Verify that the wallet holds a valid HMAC_NAME for the contact name
460 if (!address_book_verify_hmac_name(&EDIT_IDENTIFIER.identity.bip32_path,
461 EDIT_IDENTIFIER.identity.gid,
462 EDIT_IDENTIFIER.identity.contact_name,
463 ctx.hmac_name)) {
464 PRINTF("[Edit Identifier] HMAC_NAME verification failed\n");
465 return SWO_SECURITY_CONDITION_NOT_SATISFIED;
466 }
467
468 // Verify that the wallet holds a valid HMAC_REST for the previous identifier
469 if (!address_book_verify_hmac_rest(&EDIT_IDENTIFIER.identity.bip32_path,
470 EDIT_IDENTIFIER.identity.gid,
471 EDIT_IDENTIFIER.identity.scope,
472 EDIT_IDENTIFIER.previous_identifier,
473 EDIT_IDENTIFIER.previous_identifier_len,
474 EDIT_IDENTIFIER.identity.blockchain_family,
475 EDIT_IDENTIFIER.identity.chain_id,
476 ctx.hmac_rest)) {
477 PRINTF("[Edit Identifier] HMAC_REST verification failed\n");
478 return SWO_SECURITY_CONDITION_NOT_SATISFIED;
479 }
480
481 // Notify the coin app so it can validate and store display data
482 if (!handle_check_edit_identifier(&EDIT_IDENTIFIER)) {
483 PRINTF("[Edit Identifier] Rejected by coin application\n");
484 return SWO_WRONG_PARAMETER_VALUE;
485 }
486
487 // Display confirmation UI
488 ui_display();
489 return SWO_NO_RESPONSE;
490}
