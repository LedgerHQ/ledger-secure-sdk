184static const nbgl_content_t *getContentAtIdx(const nbgl_genericContents_t *genericContents,
185 int8_t contentIdx,
186 nbgl_content_t *content)
187{
188 if (contentIdx < 0 || contentIdx >= genericContents->nbContents) {
189 LOG_DEBUG(USE_CASE_LOGGER, "No content available at %d\n", contentIdx);
190 return NULL;
191 }
192
193 if (genericContents->callbackCallNeeded) {
194 if (content == NULL) {
195 LOG_DEBUG(USE_CASE_LOGGER, "Invalid content variable\n");
196 return NULL;
197 }
198 // Retrieve content through callback, but first memset the content.
199 memset(content, 0, sizeof(nbgl_content_t));
200 genericContents->contentGetterCallback(contentIdx, content);
201 return content;
202 }
203 else {
204 // Retrieve content through list
205 return PIC(&genericContents->contentsList[contentIdx]);
206 }
207}

209// Helper to retrieve the content inside a nbgl_genericContents_t using
210// either the contentsList or using the contentGetterCallback
211static const nbgl_content_t *getContentElemAtIdx(uint8_t elemIdx,
212 uint8_t *elemContentIdx,
213 nbgl_content_t *content)
214{
215 const nbgl_genericContents_t *genericContents = NULL;
216 const nbgl_content_t *p_content = NULL;
217 uint8_t nbPages = 0;
218 uint8_t elemNbPages = 0;
219
220 switch (context.type) {
221 case SETTINGS_USE_CASE:
222 case HOME_USE_CASE:
223 case GENERIC_SETTINGS:
224 genericContents = context.home.settingContents;
225 break;
226 case CONTENT_USE_CASE:
227 case GENERIC_REVIEW_USE_CASE:
228 genericContents = &context.content.genericContents;
229 break;
230 default:
231 return NULL;
232 }
233 for (int i = 0; i < genericContents->nbContents; i++) {
234 p_content = getContentAtIdx(genericContents, i, content);
235 elemNbPages = getContentNbElement(p_content);
236 if (nbPages + elemNbPages > elemIdx) {
237 *elemContentIdx = context.currentPage - nbPages;
238 break;
239 }
240 nbPages += elemNbPages;
241 }
242
243 return p_content;
244}

497static bool buttonGenericCallback(nbgl_buttonEvent_t event, nbgl_stepPosition_t *pos)
498{
499 uint8_t elemIdx;
500 uint8_t token = 0;
501 uint8_t index = 0;
502 const nbgl_content_t *p_content = NULL;
503 nbgl_content_t content = {0};
504
505 if (event == BUTTON_LEFT_PRESSED) {
506 if (context.currentPage > 0) {
507 context.currentPage--;
508 }
509 // in streaming+skip case, it is allowed to go backward at the first tag/value, except for
510 // the first set
511 else if ((context.type != STREAMING_CONTINUE_REVIEW_USE_CASE)
512 || (context.review.skipCallback == NULL) || (context.review.nbDataSets == 1)) {
513 // Drop the event
514 return false;
515 }
516 *pos = BACKWARD_DIRECTION;
517 }
518 else if (event == BUTTON_RIGHT_PRESSED) {
519 if (context.currentPage < (int) (context.nbPages - 1)) {
520 context.currentPage++;
521 }
522 else {
523 // Drop the event
524 return false;
525 }
526 *pos = FORWARD_DIRECTION;
527 }
528 else {
529 if (event == BUTTON_BOTH_PRESSED) {
530 if (context.stepCallback != NULL) {
531 context.stepCallback();
532 }
533 else if ((context.type == CONTENT_USE_CASE)
534 || (context.type == GENERIC_REVIEW_USE_CASE)) {
535 p_content = getContentElemAtIdx(context.currentPage, &elemIdx, &content);
536 if (p_content != NULL) {
537 switch (p_content->type) {
538 case CENTERED_INFO:
539 // No associated callback
540 return false;
541 case INFO_BUTTON:
542 token = p_content->content.infoButton.buttonToken;
543 break;
544 case SWITCHES_LIST:
545 token = p_content->content.switchesList.switches->token;
546 break;
547 case BARS_LIST:
548 token = p_content->content.barsList.tokens[context.currentPage];
549 break;
550 case CHOICES_LIST:
551 token = p_content->content.choicesList.token;
552 index = context.currentPage;
553 break;
554 default:
555 break;
556 }
557
558 if ((p_content) && (p_content->contentActionCallback != NULL)) {
559 p_content->contentActionCallback(token, 0, context.currentPage);
560 }
561 else if (context.content.controlsCallback != NULL) {
562 context.content.controlsCallback(token, index);
563 }
564 }
565 }
566 }
567 return false;
568 }
569 return true;
570}

721// function used to display the current page in review
722static void displayReviewPage(nbgl_stepPosition_t pos)
723{
724 uint8_t reviewPages = 0;
725 uint8_t finalPages = 0;
726 uint8_t pairIndex = 0;
727 const char *text = NULL;
728 const char *subText = NULL;
729 const nbgl_icon_details_t *icon = NULL;
730 uint8_t currentIndex = 0;
731 uint8_t warnIndex = 255;
732 uint8_t titleIndex = 255;
733 uint8_t subIndex = 255;
734 uint8_t approveIndex = 255;
735 uint8_t rejectIndex = 255;
736
737 context.stepCallback = NULL;
738
739 // Determine the 1st page to display tag/values
740 if (context.type == REVIEW_BLIND_SIGN_USE_CASE) {
741 // Warning page to display
742 warnIndex = currentIndex++;
743 reviewPages++;
744 }
745 // Title page to display
746 titleIndex = currentIndex++;
747 reviewPages++;
748 if (context.review.reviewSubTitle) {
749 // subtitle page to display
750 subIndex = currentIndex++;
751 reviewPages++;
752 }
753 approveIndex = context.nbPages - 2;
754 rejectIndex = context.nbPages - 1;
755 finalPages = approveIndex;
756
757 // Determine which page to display
758 if (context.currentPage >= finalPages) {
759 if (context.currentPage == approveIndex) {
760 // Approve page
761 icon = &C_icon_validate_14;
762 text = "Approve";
763 context.stepCallback = onReviewAccept;
764 }
