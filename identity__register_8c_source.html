208static bool verify_fields(const s_identity_ctx *context)
209{
210 bool result = TLV_CHECK_RECEIVED_TAGS(context->received_tags,
211 TAG_STRUCTURE_TYPE,
212 TAG_STRUCTURE_VERSION,
213 TAG_CONTACT_NAME,
214 TAG_SCOPE,
215 TAG_ACCOUNT_IDENTIFIER,
216 TAG_DERIVATION_PATH,
217 TAG_BLOCKCHAIN_FAMILY);
218 if (!result) {
219 PRINTF("Missing mandatory fields in Register Identity descriptor!\n");
220 return false;
221 }
222 if (context->identity->blockchain_family == FAMILY_ETHEREUM) {
223 result = TLV_CHECK_RECEIVED_TAGS(context->received_tags, TAG_CHAIN_ID);
224 if (!result) {
225 PRINTF("Missing CHAIN_ID for Ethereum family in Register Identity descriptor!\n");
226 return false;
227 }
228 }
229 return true;
230}

274static bool build_and_send_response(void)
275{
276 uint8_t group_handle[GROUP_HANDLE_SIZE] = {0};
277 uint8_t hmac_name[CX_SHA256_SIZE] = {0};
278 uint8_t hmac_rest[CX_SHA256_SIZE] = {0};
279 bool ok = false;
280
281 if (!address_book_generate_group_handle(&IDENTITY.bip32_path, group_handle)) {
282 PRINTF("[Register Identity] Error: Failed to generate group handle\n");
283 goto end;
284 }
285 // gid is the first GID_SIZE bytes of group_handle
286 if (!address_book_compute_hmac_name(
287 &IDENTITY.bip32_path, group_handle, IDENTITY.contact_name, hmac_name)) {
288 PRINTF("[Register Identity] Error: Failed to compute HMAC_NAME\n");
289 goto end;
290 }
291 if (!address_book_compute_hmac_rest(&IDENTITY.bip32_path,
292 group_handle,
293 IDENTITY.scope,
294 IDENTITY.identifier,
295 IDENTITY.identifier_len,
296 IDENTITY.blockchain_family,
297 IDENTITY.chain_id,
298 hmac_rest)) {
299 PRINTF("[Register Identity] Error: Failed to compute HMAC_REST\n");
300 goto end;
301 }
302 ok = address_book_send_register_identity_response(group_handle, hmac_name, hmac_rest);
303
304end:
305 explicit_bzero(group_handle, sizeof(group_handle));
306 explicit_bzero(hmac_name, sizeof(hmac_name));
307 explicit_bzero(hmac_rest, sizeof(hmac_rest));
308 return ok;
309}

316static void review_choice(bool confirm)

362bolos_err_t register_identity(uint8_t *buffer_in, size_t buffer_in_length)
363{
364 const buffer_t payload = {.ptr = buffer_in, .size = buffer_in_length};
365 s_identity_ctx ctx = {0};
366
367 // Init the structure
368 ctx.identity = &IDENTITY;
369 memset(&IDENTITY, 0, sizeof(IDENTITY));
370
371 // Parse using SDK TLV parser
372 if (!identity_tlv_parser(&payload, &ctx, &ctx.received_tags)) {
373 PRINTF("[Register Identity] Failed to parse TLV payload\n");
374 return SWO_INCORRECT_DATA;
375 }
376 if (!verify_fields(&ctx)) {
377 return SWO_INCORRECT_DATA;
378 }
379 print_payload(&ctx);
380
381 // Check the Identity validity according to the Coin application logic
382 if (!handle_check_identity(ctx.identity)) {
383 PRINTF("[Register Identity] Error: Identity rejected by coin application\n");
384 return SWO_WRONG_PARAMETER_VALUE;
385 }
386
387 // Display confirmation UI
388 ui_display();
389 return SWO_NO_RESPONSE;
390}
