273static bool verify_fields(const s_provide_contact_ctx *context)
274{
275 bool result = TLV_CHECK_RECEIVED_TAGS(context->received_tags,
276 TAG_STRUCTURE_TYPE,
277 TAG_STRUCTURE_VERSION,
278 TAG_CONTACT_NAME,
279 TAG_SCOPE,
280 TAG_ACCOUNT_IDENTIFIER,
281 TAG_GROUP_HANDLE,
282 TAG_DERIVATION_PATH,
283 TAG_BLOCKCHAIN_FAMILY,
284 TAG_HMAC_NAME,
285 TAG_HMAC_REST);
286 if (!result) {
287 PRINTF("[Provide Contact] Missing mandatory fields!\n");
288 return false;
289 }
290 if (context->identity->blockchain_family == FAMILY_ETHEREUM) {
291 result = TLV_CHECK_RECEIVED_TAGS(context->received_tags, TAG_CHAIN_ID);
292 if (!result) {
293 PRINTF("[Provide Contact] Missing CHAIN_ID for Ethereum family!\n");
294 return false;
295 }
296 }
297 return true;
298}

332bolos_err_t provide_contact(uint8_t *buffer_in, size_t buffer_in_length)
333{
334 const buffer_t payload = {.ptr = buffer_in, .size = buffer_in_length};
335 s_provide_contact_ctx ctx = {0};
336
337 // Init the structure
338 ctx.identity = &PROVIDE_CONTACT_IDENTITY;
339 memset(&PROVIDE_CONTACT_IDENTITY, 0, sizeof(PROVIDE_CONTACT_IDENTITY));
340
341 // Parse using SDK TLV parser
342 if (!provide_contact_tlv_parser(&payload, &ctx, &ctx.received_tags)) {
343 PRINTF("[Provide Contact] TLV parsing failed\n");
344 return SWO_INCORRECT_DATA;
345 }
346 if (!verify_fields(&ctx)) {
347 return SWO_INCORRECT_DATA;
348 }
349 print_payload(&ctx);
350
351 // Verify the group handle and extract the gid
352 if (!address_book_verify_group_handle(
353 &PROVIDE_CONTACT_IDENTITY.bip32_path, ctx.group_handle, PROVIDE_CONTACT_IDENTITY.gid)) {
354 PRINTF("[Provide Contact] Group handle verification failed\n");
355 return SWO_SECURITY_CONDITION_NOT_SATISFIED;
356 }
357
358 // Verify HMAC_PROOF over (gid, contact_name)
359 if (!address_book_verify_hmac_name(&PROVIDE_CONTACT_IDENTITY.bip32_path,
360 PROVIDE_CONTACT_IDENTITY.gid,
361 PROVIDE_CONTACT_IDENTITY.contact_name,
362 ctx.hmac_proof)) {
363 PRINTF("[Provide Contact] HMAC_PROOF verification failed\n");
364 return SWO_SECURITY_CONDITION_NOT_SATISFIED;
365 }
366
367 // Verify HMAC_REST over (gid, scope, identifier, family[, chain_id])
368 if (!address_book_verify_hmac_rest(&PROVIDE_CONTACT_IDENTITY.bip32_path,
369 PROVIDE_CONTACT_IDENTITY.gid,
370 PROVIDE_CONTACT_IDENTITY.scope,
371 PROVIDE_CONTACT_IDENTITY.identifier,
372 PROVIDE_CONTACT_IDENTITY.identifier_len,
373 PROVIDE_CONTACT_IDENTITY.blockchain_family,
374 PROVIDE_CONTACT_IDENTITY.chain_id,
375 ctx.hmac_rest)) {
376 PRINTF("[Provide Contact] HMAC_REST verification failed\n");
377 return SWO_SECURITY_CONDITION_NOT_SATISFIED;
378 }
379
380 // Pass verified contact to the coin app for storage
381 if (!handle_provide_contact(&PROVIDE_CONTACT_IDENTITY)) {
382 PRINTF("[Provide Contact] Rejected by coin application\n");
383 return SWO_WRONG_PARAMETER_VALUE;
384 }
385
386 return SWO_SUCCESS;
387}
