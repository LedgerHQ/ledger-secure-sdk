172static bool verify_fields(const s_ledger_account_ctx *context)
173{
174 bool result = TLV_CHECK_RECEIVED_TAGS(context->received_tags,
175 TAG_STRUCTURE_TYPE,
176 TAG_STRUCTURE_VERSION,
177 TAG_CONTACT_NAME,
178 TAG_DERIVATION_PATH,
179 TAG_BLOCKCHAIN_FAMILY);
180 if (!result) {
181 PRINTF("Missing mandatory fields in Ledger Account descriptor!\n");
182 return false;
183 }
184 if (context->ledger_account->blockchain_family == FAMILY_ETHEREUM) {
185 result = TLV_CHECK_RECEIVED_TAGS(context->received_tags, TAG_CHAIN_ID);
186 if (!result) {
187 PRINTF("Missing CHAIN_ID for Ethereum family in Ledger Account descriptor!\n");
188 return false;
189 }
190 }
191 return true;
192}

234static bool build_and_send_response(void)
235{
236 uint8_t hmac_proof[CX_SHA256_SIZE] = {0};
237
238 if (!address_book_compute_hmac_proof_ledger_account(&LEDGER_ACCOUNT.bip32_path,
239 (const char *) LEDGER_ACCOUNT.account_name,
240 LEDGER_ACCOUNT.blockchain_family,
241 LEDGER_ACCOUNT.chain_id,
242 hmac_proof)) {
243 PRINTF("[Ledger Account] Error: Failed to compute HMAC proof\n");
244 return false;
245 }
246
247 bool ok = address_book_send_hmac_proof(TYPE_REGISTER_LEDGER_ACCOUNT, hmac_proof);
248 explicit_bzero(hmac_proof, sizeof(hmac_proof));
249 return ok;
250}

285bolos_err_t register_ledger_account(uint8_t *buffer_in, size_t buffer_in_length)
286{
287 const buffer_t payload = {.ptr = buffer_in, .size = buffer_in_length};
288 s_ledger_account_ctx ctx = {0};
289
290 // Init the structure
291 ctx.ledger_account = &LEDGER_ACCOUNT;
292 memset(&LEDGER_ACCOUNT, 0, sizeof(LEDGER_ACCOUNT));
293
294 // Parse using SDK TLV parser
295 if (!ledger_account_tlv_parser(&payload, &ctx, &ctx.received_tags)) {
296 PRINTF("[Ledger Account] Failed to parse TLV payload\n");
297 return SWO_INCORRECT_DATA;
298 }
299 if (!verify_fields(&ctx)) {
300 return SWO_INCORRECT_DATA;
301 }
302 print_payload(&ctx);
303
304 // Check the account validity according to the Coin application logic
305 if (!handle_check_ledger_account(ctx.ledger_account)) {
306 PRINTF("[Ledger Account] Error: Account rejected by coin application\n");
307 return SWO_WRONG_PARAMETER_VALUE;
308 }
309
310 // Display confirmation UI (delegated to coin app)
311 display_register_ledger_account_review(review_choice);
312 return SWO_NO_RESPONSE;
313}
