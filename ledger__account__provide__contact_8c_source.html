197static bool verify_fields(const s_provide_ledger_account_ctx *context)
198{
199 bool result = TLV_CHECK_RECEIVED_TAGS(context->received_tags,
200 TAG_STRUCTURE_TYPE,
201 TAG_STRUCTURE_VERSION,
202 TAG_CONTACT_NAME,
203 TAG_DERIVATION_PATH,
204 TAG_BLOCKCHAIN_FAMILY,
205 TAG_HMAC_PROOF);
206 if (!result) {
207 PRINTF("[Provide Ledger Account] Missing mandatory fields!\n");
208 return false;
209 }
210 if (context->ledger_account->blockchain_family == FAMILY_ETHEREUM) {
211 result = TLV_CHECK_RECEIVED_TAGS(context->received_tags, TAG_CHAIN_ID);
212 if (!result) {
213 PRINTF("[Provide Ledger Account] Missing CHAIN_ID for Ethereum family!\n");
214 return false;
215 }
216 }
217 return true;
218}

263bolos_err_t provide_ledger_account_contact(uint8_t *buffer_in, size_t buffer_in_length)
264{
265 const buffer_t payload = {.ptr = buffer_in, .size = buffer_in_length};
266 s_provide_ledger_account_ctx ctx = {0};
267
268 // Init the structure
269 ctx.ledger_account = &PROVIDE_LEDGER_ACCOUNT;
270 memset(&PROVIDE_LEDGER_ACCOUNT, 0, sizeof(PROVIDE_LEDGER_ACCOUNT));
271
272 // Parse using SDK TLV parser
273 if (!provide_ledger_account_tlv_parser(&payload, &ctx, &ctx.received_tags)) {
274 PRINTF("[Provide Ledger Account] TLV parsing failed\n");
275 return SWO_INCORRECT_DATA;
276 }
277 if (!verify_fields(&ctx)) {
278 return SWO_INCORRECT_DATA;
279 }
280 print_payload(&ctx);
281
282 // Verify HMAC Proof of Registration: HMAC(bip32_path, account_name, family, chain_id)
283 if (!address_book_verify_hmac_proof_ledger_account(&PROVIDE_LEDGER_ACCOUNT.bip32_path,
284 PROVIDE_LEDGER_ACCOUNT.account_name,
285 PROVIDE_LEDGER_ACCOUNT.blockchain_family,
286 PROVIDE_LEDGER_ACCOUNT.chain_id,
287 ctx.hmac_proof)) {
288 PRINTF("[Provide Ledger Account] HMAC proof verification failed\n");
289 return SWO_SECURITY_CONDITION_NOT_SATISFIED;
290 }
291
292 // Pass verified contact to the coin app for storage
293 if (!handle_provide_ledger_account_contact(&PROVIDE_LEDGER_ACCOUNT)) {
294 PRINTF("[Provide Ledger Account] Rejected by coin application\n");
295 return SWO_WRONG_PARAMETER_VALUE;
296 }
297
298 return SWO_SUCCESS;
299}
